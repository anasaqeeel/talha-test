\chapter{Executive Summary}

\begin{summarybox}
This report documents progress on the Crypto Sentry project against the client's development guide. The majority of the doc has been implemented: two-server architecture (\S4, \S7), full setup and environment (\S6), database with code-first migrations (\S13), Express surveillance engine with 30s polling and 2\% flash-crash detection with idempotency (\S11, \S15), NextAuth with Credentials and optional Google (\S14), all API endpoints from the doc (\S9), and the full Next.js terminal UI including dashboard with real-time polling and status badges (\S12), watchlist (\S16), alerts with filter (\S12.1), market page with search (\S12.1), and operator onboarding with spotlight guide (\S17). \end{summarybox}

% ==================== CHAPTER 2: COMPLETED WORK ====================
\chapter{Completed Work -- Detailed}

\section{Project Setup and Environment (\S3, \S6)}

\begin{itemize}[leftmargin=*]
  \item \textbf{Stack:} Next.js 16 (App Router), React 19, TypeScript, Tailwind CSS 4, Framer Motion, Lucide React; Express 5; Prisma; PostgreSQL; NextAuth (Auth.js v5); CoinGecko API.
  \item \textbf{Env:} \texttt{.env.local} holds \texttt{DATABASE\_URL}, \texttt{AUTH\_SECRET}, \texttt{NEXTAUTH\_SECRET}, \texttt{NEXTAUTH\_URL}, \texttt{SURVEILLANCE\_PORT}, \texttt{NEXT\_PUBLIC\_SURVEILLANCE\_URL}; optional \texttt{GOOGLE\_CLIENT\_ID}, \texttt{GOOGLE\_CLIENT\_SECRET}.
  \item \textbf{Scripts (package.json):} \texttt{dev}, \texttt{dev:server}, \texttt{dev:all} (concurrently both), \texttt{db:migrate}, \texttt{db:push}, \texttt{db:generate}, \texttt{db:studio}, \texttt{db:seed}.
  \item \textbf{Prisma and env:} \texttt{scripts/prisma-env.js} loads \texttt{.env.local} and runs Prisma CLI so all DB commands use the same env.
\end{itemize}

\section{Database (\S13)}

\begin{itemize}[leftmargin=*]
  \item \textbf{Approach:} Code-first. Single source of truth: \texttt{prisma/schema.prisma}.
  \item \textbf{User:} \texttt{id} (cuid), \texttt{email} (unique), \texttt{name}, \texttt{password} (optional for Google-only), \texttt{createdAt}, \texttt{updatedAt}; relations to Watchlist and CryptoAlert.
  \item \textbf{Watchlist:} \texttt{id}, \texttt{coinId}, \texttt{coinName}, \texttt{symbol}, \texttt{userId} (FK $\to$ User, onDelete Cascade), \texttt{createdAt}; \texttt{@@unique([userId, coinId])}.
  \item \textbf{CryptoAlert:} \texttt{id}, \texttt{coinId}, \texttt{coinName}, \texttt{symbol}, \texttt{dropPercent}, \texttt{priceBefore}, \texttt{priceAfter}, \texttt{triggeredAt}, \texttt{userId} (optional FK).
  \item \textbf{Migrations:} \texttt{npm run db:migrate} generates/applies SQL under \texttt{prisma/migrations/}. Viewable via PgAdmin (database \texttt{crypto\_sentry}) or \texttt{npm run db:studio}.
  \item \textbf{Seed:} \texttt{prisma/seed.ts} creates demo user \texttt{demo@cryptosentry.com} / \texttt{demo1234}.
  \item \textbf{Client:} \texttt{src/lib/prisma.ts} exports singleton PrismaClient (\S13.6).
\end{itemize}

\section{Express Surveillance Engine (\S7, \S10, \S11)}

\begin{itemize}[leftmargin=*]
  \item \textbf{File:} \texttt{server.js} (Node, port 4000). Dependencies: express, cors, axios, @prisma/client, dotenv. CORS allows localhost:3000, 3002, 3003.
  \item \textbf{MemoryCache (\S11.4):} In-memory store for \texttt{prices}, \texttt{baselines}, \texttt{alerts} (last 100). Methods: \texttt{updatePrice}, \texttt{getAll}, \texttt{getAlerts(limit)}, \texttt{addAlert}, \texttt{getActiveAlertCoinIds()} (coins in ``alert'' in last 60s).
  \item \textbf{FlashCrashDetector (\S11.3, \S15):} Baseline per coin; drop $\ge$ 2\% $\to$ idempotency check (Prisma: no CryptoAlert same coin in 60s) $\to$ \texttt{prisma.cryptoAlert.create} and \texttt{cache.addAlert}; baseline updated.
  \item \textbf{Coin list:} Array \texttt{COINS} (15 CoinGecko ids). Single batched request per cycle (\S10.2).
  \item \textbf{fetchWithRetry (\S11.2):} GET to CoinGecko; on 429/error, exponential backoff (1s, 2s, 4s), retry up to 3 times.
  \item \textbf{Polling:} \texttt{fetchMarketData()} on startup and \texttt{setInterval(fetchMarketData, 30000)} (\S10.2).
  \item \textbf{Routes:} \texttt{GET /api/prices} (cache + \texttt{status: `stable'|`alert'}), \texttt{GET /api/alerts} (\texttt{?limit=50}), \texttt{GET /api/health}, \texttt{GET /cache}.
\end{itemize}

\section{Authentication and Authorization (\S14)}

\begin{itemize}[leftmargin=*]
  \item \textbf{NextAuth:} \texttt{src/auth.ts}. Providers: Credentials (email/password), Google (optional).
  \item \textbf{Credentials authorize (\S14.4--14.5):} \texttt{prisma.user.findUnique} by email; \texttt{bcrypt.compare}; return \texttt{\{ id, email, name \}}; null if no password (Google-only). Generic ``invalid credentials'' (\S14.10).
  \item \textbf{Google:} JWT callback upserts User by email; \texttt{token.id} = DB user id.
  \item \textbf{Session:} JWT strategy (\S14.2); cookie-based; custom \texttt{/login} (\S14.3). Middleware protects \texttt{(app)} routes (\S14.6).
  \item \textbf{Signup:} \texttt{POST /api/signup} in \texttt{src/app/api/signup/route.ts}: validate; \texttt{bcrypt.hash(password, 12)}; \texttt{prisma.user.create}; 201 or 409 (\S14.4).
\end{itemize}

\section{Data Flows (\S8)}

\begin{itemize}[leftmargin=*]
  \item \textbf{Auth (\S8.1):} Login form $\to$ NextAuth signIn $\to$ authorize $\to$ JWT cookie $\to$ protected routes read session.
  \item \textbf{Dashboard (\S8.2):} Page load $\to$ client fetches \texttt{/api/prices} and \texttt{/api/watchlist}; \texttt{setInterval(fetch, 5000)}; \texttt{clearInterval} on unmount (\S12.4). Response includes status per coin; UI shows STABLE/alert and stale warning if data $>$60s.
  \item \textbf{Watchlist (\S8.3, \S16):} Star click $\to$ POST/DELETE watchlist; server CRUD; watchlist page shows user's starred coins.
\end{itemize}

\section{Next.js UI -- Pages and Components (\S12.1)}

\begin{itemize}[leftmargin=*]
  \item \textbf{Layouts/pages:} \texttt{app/layout.tsx} (SessionProvider); \texttt{app/page.tsx} (redirect); \texttt{login}, \texttt{signup}; \texttt{(app)/layout.tsx} (auth, Sidebar); \texttt{dashboard} (5s poll, AnimatedPriceCard, StaleDataBanner, OnboardingSpotlight), \texttt{watchlist}, \texttt{alerts} (filter by asset), \texttt{market} (search).
  \item \textbf{Sidebar:} ids \texttt{nav-dashboard}, \texttt{nav-market}, \texttt{nav-watchlist}, \texttt{nav-alerts}, \texttt{nav-profile}; alert badge; signOut to \texttt{/login}.
  \item \textbf{AnimatedPriceCard (\S12.4):} STABLE (green) / ``Protocol violation'' (red) badge; border and glow; price; watchlist star; flash on tick.
  \item \textbf{StaleDataBanner (\S12.4):} When lastUpdate $>$60s or offline; refresh button.
  \item \textbf{OnboardingSpotlight (\S17):} Steps: Welcome, dashboard-stats, nav-alerts, watchlist, market, profile; \texttt{getBoundingClientRect} and clip-path; localStorage; Skip/Next/Finish.
\end{itemize}

\section{Security and Validation (\S12.3, \S14)}

\begin{itemize}[leftmargin=*]
  \item All watchlist endpoints require session; DELETE [id] verifies entry belongs to \texttt{session.user.id} (\S14.7).
  \item Signup: email and password required; duplicate email 409; bcrypt rounds 12.
  \item Credentials login: generic ``invalid credentials''; \texttt{user.password} null check for Google-only.
\end{itemize}

% ==================== CHAPTER 4: HOW TO RUN ====================
\chapter{How to Run}

\begin{lstlisting}
# Database (once)
createdb crypto_sentry
# .env.local: DATABASE_URL, AUTH_SECRET, NEXTAUTH_SECRET, NEXTAUTH_URL, etc.

npm run db:migrate    # create/update tables
npm run db:seed       # optional: demo@cryptosentry.com / demo1234

# Application
npm run dev:all       # Express on 4000, Next.js on 3000/3003
\end{lstlisting}

\textbf{Surveillance engine:} \url{http://localhost:4000} (health: \texttt{/api/health}). \\
\textbf{App:} \url{http://localhost:3003} (or port in terminal); set \texttt{NEXTAUTH\_URL} to that port. \\
\textbf{DB UI:} \texttt{npm run db:studio} or PgAdmin $\to$ database \texttt{crypto\_sentry}.
